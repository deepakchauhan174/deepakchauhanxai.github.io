const canvas = document.getElementById("glowEffect"); const ctx = canvas.getContext("2d"); canvas.width = window.innerWidth; canvas.height = window.innerHeight; let particles = []; for (let i = 0; i < 120; i++) { particles.push({ x: Math.random() * canvas.width, y: Math.random() * canvas.height, size: Math.random() * 2 + 1, velocityX: (Math.random() - 0.5) * 0.7, velocityY: (Math.random() - 0.5) * 0.7 }); } function animateParticles() { ctx.clearRect(0, 0, canvas.width, canvas.height); particles.forEach(p => { ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2); ctx.fillStyle = "#00ffffaa"; ctx.fill(); p.x += p.velocityX; p.y += p.velocityY; if (p.x < 0 || p.x > canvas.width) p.velocityX *= -1; if (p.y < 0 || p.y > canvas.height) p.velocityY *= -1; }); requestAnimationFrame(animateParticles); } animateParticles();
